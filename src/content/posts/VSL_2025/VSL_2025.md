    token = re.search(r'name="_csrf" value="([^"]+)"', r.text)
    if token:
        return token.group(1)
    meta = re.search(r'<meta name="_csrf" content="([^"]+)"', r.text)
    if meta:
        return meta.group(1)
    print("[x] CSRF Token Not Found!")
    sys.exit(1)

        "username": username,

        "_csrf": csrf
